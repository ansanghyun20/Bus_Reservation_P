```jsp
<html>
<head>
   <title>Welcome</title>
</head>
<body>
  <%!
    int count = 3;
    String makeItLower(String data){
    return data.toUpperCase();
    }
  %>
  <%
    for(int i = 1 ; i<=count; i++)
    out.println("Java Server Pages" + i +".<br>");
  %>
  <%= makeItLower("Hello World") %>
</body>
</html>
```

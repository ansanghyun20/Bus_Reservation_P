# 퀴즈
　    
### 3번문제 함수로 사칙연산 구현 및 switch문 사용
　     
      
```java
//-----------------3번문제  4칙연산 만들기

import java.util.Scanner;

public class Main {
	public float add(float a, float b) {
		return a + b;
	}

	public float div(float a, float b) {
		return a / b;
	}

	public float mul(float a, float b) {
		return a * b;
	}

	public float min(float a, float b) {
		return a - b;
	}

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		Main Calc = new Main();
		int a, b;
		String iter;
		float p = 0;
		while (true) {
			a = sc.nextInt();
			iter = sc.next();
			b = sc.nextInt();
			switch (iter) {
				case "+":
					p = Calc.add(a, b);
					break;
				case "/":
					p = Calc.div(a, b);
					break;
				case "*":
					p = Calc.mul(a, b);
					break;
				case "-":
					p = Calc.min(a, b);
					break;
			}

			System.out.println(a+" "+ iter + " "+  b +" = "+ p);
		}
	}
}
```

```java
	public static void main(String[] args) {
		for(int i =0; i<=10; i++) {
```

```java
import java.util.Scanner;
public class test5 {
	public static void main(String[] args) {
		boolean run = true;
		int a=0;
		int balance = 0;
		int money=0;
		Scanner scanner = new Scanner(System.in);
		
		while(run) {
			System.out.println("-----------------------------------");
			System.out.println("1. 예금 | 2. 출금 | 3. 잔고 | 4. 종료");
			System.out.println("-----------------------------------");
			System.out.print("선택> ");
			a= scanner.nextInt();
			switch(a) {
				case 1: System.out.print("예금액> ");
						balance= scanner.nextInt();
						money = money+balance;
						System.out.println("");
						break;

				case 2: System.out.print("출금액> ");
						balance= scanner.nextInt();
						money= money - balance;
						System.out.println("");
						break;
				case 3: System.out.println("예금액: " + money);
						System.out.println("");
						break;
				case 4 : System.exit(0);
			}
		}
	}
}
```

Amoeba -> 아메바는 상속을 받았지만 재정의를 하면서 출력 결과 값이 달라지게 된다.
```java
//Amoeba
public class Amoeba extends shape{
	@Override
	public void rotate() {
		System.out.println("아메바 답게 돌아요");
	}
	@Override
	public void playSound() {
		System.out.println("아메바다운 소리를 내어요");
	}
}
```
Main
```java
//Main
public class ShapeMain {
	public static void main(String[] args) {
		Square s = new Square();
		s.rotate();
		s.playSound();

		Amoeba a = new Amoeba();
		a.rotate();
	}
}
```
